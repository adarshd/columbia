	\subsection*{28)}
		Given $x_1 < x_2$:
		\begin{align}
			F(x_2) &= P(X \leq x_2)\\
			&= P( X \leq x_2 \cup x_1 < X \leq x_2) \\
			&= P( X \leq x_2 ) + P( x_1 < X \leq x_2) \\
			&= F(x_1) + P( x_1 < X \leq x_2)\\
			& \geq F(x_1)
		\end{align}
		$F(x_1) = F(x_2) \iff P(x_1 < X \leq x_2) =0.$
